DECLARE
	CURSOR EMP_CURSOR
	(P_JOBID EMPLOYEES.JOB_ID%TYPE DEFAULT 'SA_REP') 
	IS  
	SELECT * FROM EMPLOYEES WHERE JOB_ID = P_JOBID;
	COUNTER NUMBER := 1;
	SUMSAL NUMBER := 0;
	FH DATE := SYSDATE;
	LH DATE := '1-JAN-1900';
BEGIN
    DBMS_OUTPUT.PUT_LINE(RPAD('SRNO',2) || '  ' || RPAD('FIRST_NAME',10) || '  ' || RPAD('SALARY',8));
	FOR EMP_RECORD IN EMP_CURSOR('ST_CLERK') LOOP
        SUMSAL := SUMSAL + EMP_RECORD.SALARY;
		DBMS_OUTPUT.PUT_LINE( RPAD(COUNTER,2) || '  ' || RPAD(EMP_RECORD.FIRST_NAME,10) || '  ' || RPAD(EMP_RECORD.SALARY,8));
		COUNTER := COUNTER + 1;	
	END LOOP;
	DBMS_OUTPUT.PUT_LINE(LPAD('SUM = ',16) || SUMSAL);
	DBMS_OUTPUT.PUT_LINE(CHR(10));
	COUNTER := 1;
	DBMS_OUTPUT.PUT_LINE(RPAD('SRNO',2) || '  ' || RPAD('LASTR_NAME',10) || '  ' || RPAD('HIRE_DATE',9));
	FOR EMP_RECORD IN EMP_CURSOR() LOOP
        SUMSAL := SUMSAL + EMP_RECORD.SALARY;
		DBMS_OUTPUT.PUT_LINE( RPAD(COUNTER,2) || '  ' || RPAD(EMP_RECORD.LAST_NAME,10) || '  ' || RPAD(EMP_RECORD.HIRE_DATE,8));
		COUNTER := COUNTER + 1;	
		IF FH > EMP_RECORD.HIRE_DATE THEN
            FH := EMP_RECORD.HIRE_DATE;
		END IF;
		IF LH < EMP_RECORD.HIRE_DATE THEN
            LH := EMP_RECORD.HIRE_DATE;
		END IF;
	END LOOP;
	DBMS_OUTPUT.PUT_LINE('FIRST HIRE_DATE IS ' || FH);
	DBMS_OUTPUT.PUT_LINE('LAST HIRE_DATE IS ' || LH);
	DBMS_OUTPUT.PUT_LINE(CHR(10));
	COUNTER := 1;
	DBMS_OUTPUT.PUT_LINE(RPAD('SRNO',2) || '  ' || 'EMPLOYEE_ID' || '  ' || 'DEPARTMENT_ID' || '  ' || 'MANAGER_ID');
	FOR EMP_RECORD IN EMP_CURSOR('&JID') LOOP
        SUMSAL := SUMSAL + EMP_RECORD.SALARY;
		DBMS_OUTPUT.PUT_LINE( RPAD(COUNTER,2) || '  ' || RPAD(EMP_RECORD.EMPLOYEE_ID,11) || '  ' || RPAD(EMP_RECORD.DEPARTMENT_ID,13) || '  ' || RPAD(EMP_RECORD.MANAGER_ID,10));
		COUNTER := COUNTER + 1;	
	END LOOP;
END;
/

-- OPEN CURSOR 3 TIMES.

-- FIRST TIME, PROVIDE SOME LITERAL/VALUE AS ACTUAL PARAMETER. DISPLAY SRNO, FIRST_NAME , SALARY. 
-- DISPLAY SUM OF SALARY AS SUMMARY. PASS PARAMETERS POSITIONALLY.

-- SECOND TIME, USE DEFAULT VALUE OF PARAMETER. DISPLAY SRNO , LAST_NAME , HIRE_DATE. 
-- DISPLAY FIRST AND LAST HIRE_DATE ON WHICH EMPLOYEES WERE HIRED FOR THIS JOB. PASS PAREMETERS USING NAMED ASSOCIATION STYLE.

-- THIRD TIME, USE SUBSTITUTION VARIABLE AS PARAMETER. DISPLAY SRNO , EMPLOYEE_ID , DEPARTMENT_ID , MANAGER_ID. PASS PARAMETERS USING COMBINATION STYLE.
