1 DISPLAY THE EMPLOYEE INFO. WHO GETS HIGHER SALARY THAN AVERAGE SALARY PAID
SELECT * FROM EMPLOYEES WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES);

2 DISPLAY THE EMPLOYEE INFO. WHO WERE HIRED FIRST IN COMPANY
SELECT * FROM EMPLOYEES WHERE HIRE_DATE = (SELECT MIN(HIRE_DATE) FROM EMPLOYEES);

3 DISPLAY THE EMPLOYEE INFO. WHO WERE HIRED LAST IN COMPANY
SELECT * FROM EMPLOYEES WHERE HIRE_DATE IN (SELECT MAX(HIRE_DATE) FROM EMPLOYEES);

4 DISPLAY INFO. OF EMPLOYEE WHO GETS HIGHEST COMMISSION AMOUNT
SELECT FIRST_NAME , SALARY , SALARY*COMMISSION_PCT "COMMISSION AMOUNT" FROM EMPLOYEES WHERE SALARY*COMMISSION_PCT = (SELECT MAX(SALARY*COMMISSION_PCT)FROM EMPLOYEES);

5 DISPLAY INFO. OF EMPLOYEE WHO GETS LOWEST COMMISSION AMOUNT
SELECT FIRST_NAME , SALARY , SALARY*COMMISSION_PCT "COMMISSION AMOUNT" FROM EMPLOYEES WHERE SALARY*COMMISSION_PCT = (SELECT MIN(SALARY*COMMISSION_PCT)FROM EMPLOYEES);

6 DISPLAY THOSE WHO GETS LOWER SALARY THAN AVERAGE SALARY PAID TO IT_PROG
SELECT FIRST_NAME , SALARY FROM EMPLOYEES WHERE SALARY < (SELECT AVG(SALARY) FROM EMPLOYEES WHERE JOB_ID LIKE 'IT_PROG');

7 DISPLAY THOSE WHO GETS HIGHER SALARY THAN HIGHEST SALARY PAID TO DEPARTMENT_ID 80
SELECT FIRST_NAME , SALARY , DEPARTMENT_ID FROM EMPLOYEES WHERE SALARY > (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID = 80);

8 DISPLAY THOSE WHO GETS LOWER SALARY THAN ALL EMPLOYEES OF DEPARTMENT_ID 60
SELECT FIRST_NAME , SALARY , DEPARTMENT_ID FROM EMPLOYEES WHERE SALARY < (SELECT MIN(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID = 60);

9 DISPLAY THOSE WHO GETS HIGHER SALARY THAN ANY SA_REP
SELECT FIRST_NAME , SALARY , JOB_ID FROM EMPLOYEES WHERE SALARY > (SELECT MAX(SALARY) FROM EMPLOYEES WHERE JOB_ID LIKE 'SA_REP');

10 DISPLAY THOSE WHO GETS LOWER SALARY THAN LOWEST SALARY PAID WHO REPORTS TO 149 (USE MULTI ROW COMPARISON OPERATOR)
SELECT FIRST_NAME , SALARY , MANAGER_ID FROM EMPLOYEES WHERE SALARY < (SELECT MIN(SALARY) FROM EMPLOYEES WHERE MANAGER_ID = 149);

11 DISPLAY THOSE WHO GETS HIGHER SALARY THAN HIGHEST SALARY PAID TO DEPARTMENT_ID 20 (USE MULTI ROW COMPARISON OPERATOR)
SELECT FIRST_NAME , SALARY , DEPARTMENT_ID FROM EMPLOYEES WHERE SALARY > (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID = 20);

12 DISPLAY ALL OTHER THAN THOSE WHO GETS HIGHEST SALARY IN THEIR DEPARTMENT
SELECT FIRST_NAME , SALARY , DEPARTMENT_ID FROM EMPLOYEES WHERE SALARY NOT IN (SELECT MAX(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

13 DISPLAY ALL OTHER THAN THOSE WHO GETS LOWEST SALARY FOR THEIR JOB
SELECT FIRST_NAME , SALARY , JOB_ID FROM EMPLOYEES WHERE SALARY NOT IN (SELECT MIN(SALARY) FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY ALL THOSE WHO GETS NEITHER HIGHEST NOR LOWER SALARY FOR THEIR MANAGER
SELECT FIRST_NAME , SALARY , MANAGER_ID FROM EMPLOYEES WHERE SALARY NOT IN (SELECT MAX(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID) AND SALARY NOT IN (SELECT MIN(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID);

15 DISPLAY THOSE EMPLOYEES WHO GETS HIGHER SALARY THAN THE AVERAGE SALARY PAID TO THOSE WHO WERE HIRED IN 94 OR 96. (DON'T USE LIKE)
SELECT FIRST_NAME , SALARY , HIRE_DATE FROM EMPLOYEES WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES WHERE TO_CHAR(HIRE_DATE,'RR') IN (94,96));

16 DISPLAY INFO. ABOUT LAST/LATEST HIRED IT_PROG.
SELECT * FROM EMPLOYEES WHERE (HIRE_DATE,JOB_ID) IN (SELECT MAX(HIRE_DATE) , JOB_ID FROM EMPLOYEES WHERE JOB_ID LIKE 'IT_PROG' GROUP BY JOB_ID) OR (HIRE_DATE,JOB_ID) IN (SELECT MIN(HIRE_DATE), JOB_ID FROM EMPLOYEES WHERE JOB_ID LIKE 'IT_PROG' GROUP BY JOB_ID);

17 DISPLAY EMPLOYEES OF DEPARTMENT 80 WHO GETS HIGHER SALARY THAN THE AVERAGE SALARY PAID BY COMPANY.
SELECT FIRST_NAME , SALARY , DEPARTMENT_ID FROM EMPLOYEES WHERE DEPARTMENT_ID = 80 AND SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES);

18 DISPLAY ALL IT_PROG WHO GETS LOWER SALARY THAN THE AVERAGE SALARY PAID BY COMPANY.
SELECT FIRST_NAME , SALARY , JOB_ID FROM EMPLOYEES WHERE JOB_ID LIKE 'IT_PROG' AND SALARY < (SELECT AVG(SALARY) FROM EMPLOYEES);

19 DISPLAY EMPLOYEES WHO REPORTS TO 124 WHO GETS LOWER SALARY THAN THE AVERAGE SALARY PAID BY COMPANY.
SELECT FIRST_NAME , SALARY , MANAGER_ID FROM EMPLOYEES WHERE MANAGER_ID = 124 AND SALARY < (SELECT AVG(SALARY) FROM EMPLOYEES);

20 DISPLAY ALL OTHER THAN THOSE WHO GETS HIGHEST SALARY FOR THEIR MANAGER.
SELECT FIRST_NAME , MANAGER_ID , SALARY FROM EMPLOYEES WHERE (MANAGER_ID,SALARY) NOT IN (SELECT MANAGER_ID , MAX(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID);

21 DISPLAY ALL WHO GETS HIGHEST SALARY FOR THEIR GRADE_LEVEL.
SELECT FIRST_NAME , SALARY , GRADE_LEVEL FROM 
EMPLOYEES E JOIN JOB_GRADES ON SALARY BETWEEN LOWEST_SAL AND HIGHEST_SAL 
WHERE (GRADE_LEVEL,SALARY) IN (SELECT GRADE_LEVEL , MAX(SALARY) FROM EMPLOYEES E JOIN JOB_GRADES ON SALARY BETWEEN LOWEST_SAL AND HIGHEST_SAL GROUP BY GRADE_LEVEL) 
ORDER BY GRADE_LEVEL;

22 DISPLAY ALL OTHER THAN THOSE WHO GETS LOWEST SALARY FOR THEIR GRADE_LEVEL.
SELECT FIRST_NAME , SALARY , GRADE_LEVEL FROM 
EMPLOYEES E JOIN JOB_GRADES ON SALARY BETWEEN LOWEST_SAL AND HIGHEST_SAL 
WHERE (GRADE_LEVEL,SALARY) NOT IN (SELECT GRADE_LEVEL , MIN(SALARY) FROM EMPLOYEES E JOIN JOB_GRADES ON SALARY BETWEEN LOWEST_SAL AND HIGHEST_SAL GROUP BY GRADE_LEVEL) 
ORDER BY GRADE_LEVEL;

23 DISPLAY INFO. ABOUT FIRST HIRED EMPLOYEES FOR DEPARTMENT 90
SELECT * FROM EMPLOYEES WHERE HIRE_DATE IN (SELECT MIN(HIRE_DATE) FROM EMPLOYEES WHERE DEPARTMENT_ID = 90 GROUP BY DEPARTMENT_ID)

24 DISPLAY INFO. ABOUT EMPLOYEES WHO WERE HIRED FIRST OR LAST FOR THEIR MANAGER.
SELECT * FROM EMPLOYEES WHERE HIRE_DATE IN (SELECT MIN(HIRE_DATE) FROM EMPLOYEES GROUP BY MANAGER_ID) OR HIRE_DATE IN (SELECT MAX(HIRE_DATE) FROM EMPLOYEES GROUP BY MANAGER_ID) ORDER BY MANAGER_ID;

25 DISPLAY ALL EXCEPT THOSE WHO WERE HIRED FIRST OR LAST FOR THEIR JOBS.
SELECT * FROM EMPLOYEES WHERE HIRE_DATE IN (SELECT MIN(HIRE_DATE) FROM EMPLOYEES GROUP BY JOB_ID) OR HIRE_DATE IN (SELECT MAX(HIRE_DATE) FROM EMPLOYEES GROUP BY JOB_ID) ORDER BY JOB_ID;
